48            lastName = newLine[1];

53  
54            if (dateTime.length() == 0) {
55              System.err.println("ERROR: Invalid datetime");
56              return;
57            }
58  
59            people.add(new MockPerson(firstName, lastName, dateTime,
60                email, gender, address));
61          }
62  
63          for (MockPerson person : people) {
64            System.out.println(person);
65          }
66        } catch (Exception e) {
67          System.err.println("ERROR: error reading from file " + args.get(1));
68        }
69      } else {
70        System.err.println("ERROR: mock requires one additional argument");
71      }
