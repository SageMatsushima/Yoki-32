172   private static class ReturnHandler implements TemplateViewRoute {
173     @Override
174     public ModelAndView handle(Request req, Response res) {
175 
176       // load stars data
177       if (tree.getNodeList().size() == 0) {
178         ArrayList<String> args = new ArrayList<>();
179         args.add("stars");
180         args.add("data/stars/stardata.csv");
181         csvParser.action(args);
182       }
183 
184       QueryParamsMap qm = req.queryMap();
185 
186       String k = qm.value("k");
187       String r = qm.value("r");
188       String name = qm.value("star-name");
189       String x = qm.value("x-coord");
190       String y = qm.value("y-coord");
191       String z = qm.value("z-coord");
192 
193       boolean nameFound = true;
194 
195       ImmutableMap.Builder<String, String> variables = new ImmutableMap.Builder()
196           .put("title", "Stars: Query the Database")
197           .put("k", k)
198           .put("r", r)
199           .put("name", name)
200           .put("x", x)
201           .put("y", y)
202           .put("z", z);
203 
204       if (qm.value("algorithm").equals("neighbors")) {
205         variables.put("neighbors", "checked");
206         variables.put("radius", "");
207       } else {
208         variables.put("neighbors", "");
209         variables.put("radius", "checked");
210       }
211 
212       if (qm.value("type").equals("naive_")) {
213         variables.put("kd", "");
214         variables.put("naive", "checked");
215       } else {
216         variables.put("kd", "checked");
217         variables.put("naive", "");
218       }
219 
220       if (qm.value("input").equals("star-name")) {
221         variables.put("nameSelected", "checked");
222         variables.put("coordinates", "");
223       } else {
224         variables.put("nameSelected", "");
225         variables.put("coordinates", "checked");
226       }
227 
228       try {
229         //neighbors or radius
230         String command = qm.value("algorithm");
231         // "" or naive_
232         String prefix = qm.value("type");
233 
234         if (command.equals("neighbors")) {
235           tree.setK(Integer.parseInt(k));
236           tree.setR(Double.POSITIVE_INFINITY);
237         } else {
238           tree.setK(Integer.MAX_VALUE);
239           tree.setR(Double.parseDouble(r));
240           System.out.println("here");
241         }
242 
243         if (qm.value("input").equals("star-name")) {
244           nameFound = searcher.setCoordsByName(name);
245         } else {
246           ArrayList<Double> coords = new ArrayList<>();
247 
248           coords.add(Double.parseDouble(x));
249           coords.add(Double.parseDouble(y));
250           coords.add(Double.parseDouble(z));
251 
252           tree.setCoords(coords);
253         }
254 
255         if (prefix.equals("naive_")) {
256           tree.naiveNeighbors();
257         } else {
258           tree.kdNeighbors();
259         }
260 
261         String table;
262         if (nameFound) {
263           ArrayList<Star> found = tree.getFound();
264           table = starsToTable(found);
265         } else {
266           table = "Name not found";
267         }
268 
269         variables.put("stars", table);
270 
271         return new ModelAndView(variables.build(), "query.ftl");
272       } catch (Exception e) {
273         variables.put("stars", e.toString());
274 
275         return new ModelAndView(variables.build(), "query.ftl");
276       }
277     }

279     private String starsToTable(ArrayList<Star> stars) {
280       StringBuilder table = new StringBuilder();
281       table.append("<table><tr><th>Name</th><th>Id</th><th>Coordinates</th></tr>");
282 
283       for (Star star : stars) {
284         table.append("<tr><td>" + star.getName() + "</td>"
285             + "<td>" + star.getId() + "</td>"
286             + "<td>" + star.getCoords() + "</td></tr>");
287       }
288 
289       table.append("</table>");
290       return table.toString();
291     }

298   private static class ExceptionPrinter implements ExceptionHandler {
299     @Override
300     public void handle(Exception e, Request req, Response res) {
301       res.status(500);
302       StringWriter stacktrace = new StringWriter();
303       try (PrintWriter pw = new PrintWriter(stacktrace)) {
304         pw.println("<pre>");
305         e.printStackTrace(pw);

307       }
308       res.body(stacktrace.toString());
