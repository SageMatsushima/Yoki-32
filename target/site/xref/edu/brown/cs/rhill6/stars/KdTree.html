9   /**
10   * A KdTree of Nodes.
11   * @param <N> A node extending KdNode
12   */
13  public class KdTree<N extends KdNode> {
14    private N root = null;
15    private int dim;
16    private final ArrayList<N> list = new ArrayList<>();
17  
18    // Fields for search functions
19    private int k;
20    private double r;
21    private ArrayList<Double> coords = new ArrayList<>();
22    private N nodeIgnored;
23    private final ArrayList<N> found = new ArrayList<>();
24    private AlexKDTree<N> alexKDTree;
25  
26    /**
27     * Constructs a KD tree from its list of nodes; nodes remain
28     * in the list but are assigned left and right children.
29     * @param nodeList List of nodes
30     */
31    public void listToTree(ArrayList<N> nodeList) {
32      // start by getting median along first dim, root
33      // recur down tree, alternating dimension
34      // pass down range to look for the median in
35  
36      if (list.size() == 0) {
37        root = null;
38        return;
39      }
40  
41      dim = list.get(0).getCoords().size();
42      alexKDTree = new AlexKDTree<>(nodeList, dim);
43      root = alexKDTree.getRoot();
44    }
45  
46    /**
47     * Performs a KD Tree search and prints the found nodes in order.
48     */
49    public void searchAndPrint() {
50      kdNeighbors();
51      printFound();
52    }

54    /**
55     * Performs a naive list search and prints the found nodes in order.
56     */
57    public void naiveSearchAndPrint() {
58      naiveNeighbors();
59      printFound();
60    }
61  
62    private void printFound() {
63      for (N node : found) {
64        System.out.println(node);
65      }
66    }

68    /**
69     * Performs a KD nearest neighbors search based on fields.
70     */
71    public void kdNeighbors() {
72      found.clear();
73      if (root == null || k == 0) {
74        return;
75      }
76      alexKDTree.setNeighbor(k);
77      alexKDTree.setRadius(r);
78      PriorityQueue<N> foundQ = alexKDTree.search(nodeIgnored, alexKDTree.getRoot(), coords);
79      heapToList(foundQ);
80    }
81  
82    private void heapToList(PriorityQueue<N> maxHeap) {
83      if (maxHeap == null || maxHeap.size() == 0) {
84        found.clear();
85        return;
86      }
87  
88      N maxElt = maxHeap.poll();
89      heapToList(maxHeap);
90      if (maxElt.distance(coords) <= r) {
91        found.add(maxElt);
92      }
93    }
94  
95    /**
96     * Performs a naive list search based on fields.
97     */
98    public void naiveNeighbors() {
99      // Don't add nodes with this distance or less, already added
100     double distAdded = -1;
101     double currentMin;
102     ArrayList<Integer> indices = new ArrayList<>();
103     double newDist;
104 
105     found.clear();
106 
107     while (found.size() < k && found.size() < list.size()) {
108       indices.clear();
109       currentMin = Double.POSITIVE_INFINITY;
110 
111       for (int i = 0; i < list.size(); i++) {
112         // If searching from a node, ignore that node
113         if (list.get(i).equals(nodeIgnored)) {
114           continue;
115         }
116 
117         newDist = list.get(i).distance(coords);
118         if (newDist < currentMin && newDist > distAdded && newDist <= r) {
119           // Found better node
120           currentMin = newDist;
121           indices.clear();
122           indices.add(i);
123         } else if (newDist == currentMin && newDist > distAdded && newDist <= r) {
124           // Two equally close nodes
125           indices.add(i);
126         }
127       }
128 
129       distAdded = currentMin;
130 
131       if (indices.size() == 0) {
132         // None found, finish
133         return;
134       } else if (found.size() + indices.size() <= k) {
135         // Print normally
136 
137         for (int idx : indices) {
138           found.add(list.get(idx));
139         }
140       } else {
141         // Randomize, print up to k total
142         // nodes_found + indices.size() > k
143         // indices.size() > k - nodes_found
144         Collections.shuffle(indices);
145         int holdSize = found.size();
146         for (int i = 0; i < k - holdSize; i++) {
147           found.add(list.get(indices.get(i)));
148         }
149       }
150     }
151   }
