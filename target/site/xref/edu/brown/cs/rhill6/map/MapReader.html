56    /**
57     * Reads in all the nodes from the file.
58     * @throws SQLException
59     * @throws ClassNotFoundException
60     */
61    private void node() throws SQLException, ClassNotFoundException {
62      if (conn != null) {
63        conn.close();
64      }
65  
66      Class.forName("org.sqlite.JDBC");
67      String urlToDB = "jdbc:sqlite:" + mapDataPath;
68      conn = DriverManager.getConnection(urlToDB);
69      Statement stat = conn.createStatement();
70      stat.executeUpdate("PRAGMA foreign_keys=ON");
71  
72      // Query nodes that are part of a distinct way
73      try (ResultSet rs = SQLcommands.mapReaderSQL();) {
74        while (rs.next()) {
75          String id = rs.getString("id");
76          double latitude = rs.getDouble("latitude");
77          double longitude = rs.getDouble("longitude");
78          MapNodeNode.html#MapNode">MapNode newNode = new MapNode(id, latitude, longitude);
79  
80          nodes.add(newNode);
81          nodeIdMap.put(id, newNode);
82        }
83        rs.close();
84      } catch (Exception e) {
85        System.err.println("ERROR: There was an error reading in node data");
86      }
87      tree = new AlexKDTree<MapNode>(nodes, 2);
88    }
