8   /**
9    * Parses a CSV files of stars into both an array and a kd tree of stars.
10   */
11  public class SearchNearest implements TriggerAction {
12    /**
13     * Parses a csv file of stars and stores them in Main.stars.
14     * @param args A list of arguments. Should be stars,file
15     */
16    public void action(ArrayList<String> args) {
17      if (args.size() == 3) {
18  
19        try {
20          Double lat = Double.parseDouble(args.get(1));
21          Double lon = Double.parseDouble(args.get(2));
22  
23          KdNode nearest = nearestNode(lat, lon);
24          System.out.println(nearest);
25        } catch (Exception e) {
26          System.err.println("ERROR: nearest takes two numbers");
27        }
28      } else {
29        System.err.println("ERROR: invalid number of arguments, must be in form (nearest # #)");
30      }
31    }
32  
33    /**
34     * Get the nearest node to lat/lon coordinates.
35     * @param lat A double
36     * @param lon A double
37     * @return A MapNode
38     */
39    public static MapNode nearestNode(Double lat, Double lon) {
40      ArrayList<Double> coords = new ArrayList<>();
41      coords.add(lat);
42      coords.add(lon);
43  
44      AlexKDTree<MapNode> tree = MapReader.getTree();
45      if (tree == null) {
46        return null;
47      }
48  
49      tree.setNeighbor(1);
50      tree.setRadius(Double.POSITIVE_INFINITY);
51      PriorityQueue<MapNode> found = tree.search(null, tree.getRoot(), coords);
52  
53      if (found.size() == 1) {
54        return found.peek();
55      } else {
56        return null;
57      }
58    }
59  }
