13   * KdTree class from Alex's code base.

16  public class AlexKDTree<N extends KdNode> {

66    /**
67     * This method uses recursion to create the tree and assigns
68     * locations to specific nodes.
69     *
70     * @param nodeList a list of nodes
71     * @param depth    the depth of the node within the tree
72     * @return the middle node/current node you are working on
73     */
74    public N createTree(List<N> nodeList, int depth) {
75      //Finding the index of x, y, z
76      int index = depth % this.dimensions;
77      nodeList.sort(new NodeCompare(index));
78      //Finding the middle node and setting its depth
79      int mid = (nodeList.size()) / 2;
80      N midNode = nodeList.get(mid);
81      midNode.setDepth(depth);
82  
83      //If size is 0 or 1, then reaches the base case
84      if (nodeList.size() == 1 || nodeList.size() == 0) {
85        return midNode;
86      }
87      //Recurse on the left hand side of the tree
88      if (mid > 0) {
89        midNode.setLeft(createTree(nodeList.subList(0, mid), depth + 1));
90      }
91      //Recurse on the right hand side of the tree
92      if (mid < nodeList.size() - 1) {
93        midNode.setRight(createTree(nodeList.subList(mid + 1, nodeList.size()), depth + 1));
94      }
95      return midNode;
96    }

98    /**
99     * This method find the neighbors that are closest to the input node.
100    *
101    * @param ignore the node that we ignore if given a name
102    * @param node   the current node we are looking at
103    * @param target the point of which we want to be the closest to
104    * @return the list of neighbors that are nearest
105    */
106   public PriorityQueue<N> search(N ignore, N node, ArrayList<Double> target) {
107     //create a new arraylist if we start out at the root
108     if (root == null || node == null) {
109       return null;
110     }
111     if (root == node) {
112       //nearest = new ArrayList<N>();
113       nearest = new PriorityQueue<>(new DistanceCompare(target));
114     }
115 
116     // If we searched from this node by name, don't add it
117     if (!node.equals(ignore)) {
118       // if the size of the list of neighbors is less than the numb,
119       // then we can add the node to the list
120       if (nearest.size() < neighbor) {
121         if (targetDis(node.getCoords(), target) <= radius) {
122           nearest.add(node);
123         }
124       } else {
125         //Edge case where the number of neighbors is 0
126         if (neighbor == 0) {
127           return nearest;
128         }
129         //Get the last node in the list
130         N largest = nearest.peek();
131       /*Check if the distance between the target point and the node is less that the
132       distance between the largest nodes and the target point
133       */
134         if (targetDis(target, node.getCoords())
135             < targetDis(largest.getCoords(), target)) {
136           //Remove farthest
137           nearest.poll();
138           //Add new
139           nearest.add(node);
140         } else if (targetDis(target, node.getCoords())
141             == targetDis(largest.getCoords(), target)) {
142           // Equidistant, add randomness
143           if (coinFlip.nextBoolean()) {
144             nearest.poll();
145             nearest.add(node);
146           }
147         }
148       }
149     }
150 
151     //get the index, the largest node, and the distance between the largest node and the targetPoint
152     int index = node.getDepth() % dimensions;
153 
154     double targetNeighbor;
155     if (nearest.size() > 0) {
156       N bigNode = nearest.peek();
157       targetNeighbor = targetDis(target, bigNode.getCoords());
158     } else {
159       // No max distance as no node found
160       targetNeighbor = Double.POSITIVE_INFINITY;
161     }
162 
163     /*If the target neighbor is greater or equal to the distance between teh current node
164     and the targetPoint axis then recurse on the left and right nodes if they exist
165     */
166     if (nearest.size() < neighbor || targetNeighbor
167         >= Math.abs(node.getCoords().get(index) - target.get(index))) {
168       search(ignore, (N) node.getLeft(), target);
169       search(ignore, (N) node.getRight(), target);
170     } else {
171       /*If the current node index coordinate is less than the target point index
172       and it has a right node or left node then recurse on the respective node
173        */
174       if ((node.getCoords().get(index) < target.get(index))) {
175         search(ignore, (N) node.getRight(), target);
176       } else {
177         search(ignore, (N) node.getLeft(), target);
178       }
179     }
180     return nearest;
181   }
182   /**
183    * This is is a comparator that compares the coordinates axis of different Nodes.
184    */
185   private final class NodeCompare implements Comparator<N> {
186     private int index;
187 
188     private NodeCompare(int i) {
189       this.index = i;
190     }
191 
192     @Override
193     public int compare(N a, N b) {
194       return Double.compare(a.getCoords().get(index), b.getCoords().get(index));
195     }
196   }

198   /**
199    * This is is a comparator that compares the distance of different Nodes.
200    */
201   private final class DistanceCompare implements Comparator<N> {
202     private ArrayList<Double> coor;
203 
204     private DistanceCompare(ArrayList<Double> coor) {
205       this.coor = coor;
206     }
207 
208     @Override
209     public int compare(N a, N b) {
210       return Double.compare(targetDis(b.getCoords(), coor),
211         targetDis(a.getCoords(), coor));
212     }
213   }

215   /**
216    * This method get the distance between a given node and the target point.
217    *
218    * @param node        current node you are looking at
219    * @param targetPoint the target point you are looking at
220    * @return the distance between the two points
221    */
222   public double targetDis(ArrayList<Double> node, ArrayList<Double> targetPoint) {
223     double distance = 0;
224     for (int i = 0; i < dimensions; i++) {
225       distance += Math.pow(node.get(i) - targetPoint.get(i), 2);
226     }
227     return Math.sqrt(distance);
228   }
