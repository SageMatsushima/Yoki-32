21    /**
22     * Testing route command and checking if dijkstras is reading in the right nodes and ways for small maps
23     * given coordinates
24     */
25    @Test
26    public void testRoutePoints() {
27      reader.action(file);
28      try {
29        args.set(1, "41.8");
30        args.set(2, "-71.3");
31        args.set(3, "42");
32        args.set(4, "-72");
33        search.action(args);
34        assertEquals(search.getOutput(), "/n/0 -> /n/1 : /w/0\n" +
35          "/n/1 -> /n/2 : /w/1\n" +
36          "/n/2 -> /n/5 : /w/4\n");
37        HashMap<MapNode, Graph<MapNode, MapWay>.NodeDistance> nodeDistanceMap = Main.getGraph().getNodeDistanceMap();
38        MapNode goal = Main.getGraph().getGoal();
39        Graph.NodeDistance goalNode = nodeDistanceMap.get(goal);
40  
41        assertTrue(goal.getLatitude() == 41.8206);
42        assertTrue(goal.getLongitude() == -71.4003);
43        assertEquals(goal.getId(),  "/n/5");
44  
45        Graph.NodeDistance one = nodeDistanceMap.get(goalNode.getPrevious());
46        MapNode oneNode = (MapNode) goalNode.getPrevious();
47        assertTrue(oneNode.getLatitude() == 41.8206);
48        assertTrue(oneNode.getLongitude() == -71.4);
49        assertEquals(oneNode.getId(),  "/n/2");
50        assertEquals(oneNode.getEdges().get(0).getId(),  "/w/4");
51  
52        Graph.NodeDistance two = nodeDistanceMap.get(one.getPrevious());
53        MapNode twoNode = (MapNode) one.getPrevious();
54        assertTrue(twoNode.getLatitude() == 41.8203);
55        assertTrue(twoNode.getLongitude() == -71.4);
56        assertEquals(twoNode.getEdges().get(0).getId(),  "/w/1");
57        assertEquals(twoNode.getEdges().get(1).getId(),  "/w/3");
58        assertEquals(twoNode.getId(),  "/n/1");
59  
60        Graph.NodeDistance three = nodeDistanceMap.get(two.getPrevious());
61        MapNode threeNode = (MapNode) two.getPrevious();
62        assertTrue(threeNode.getLatitude() == 41.82);
63        assertTrue(threeNode.getLongitude() == -71.4);
64        assertEquals(threeNode.getEdges().get(0).getId(),  "/w/0");
65        assertEquals(threeNode.getEdges().get(1).getId(),  "/w/2");
66        assertEquals(threeNode.getId(),  "/n/0");
67        assertTrue(three.getMinDist() == 0.0);
68  
69      } catch(Exception e) {
70        System.err.println("ERROR: Error parsing file in testNumLines");
71      }
72    }
