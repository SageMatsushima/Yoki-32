14    private MapReader reader = new MapReader();
15    private ArrayList<String> file = new ArrayList<>(Arrays.asList("map", "data/maps/smallMaps.sqlite3"));
16    private ArrayList<String> args = new ArrayList<>(Arrays.asList("route", "", "", "", ""));
17    private SearchRoute search = new SearchRoute();
18  
19    @Test
20    public void testRoutePoints() {
21      reader.action(file);
22      try {
23        args.set(1, "41.8");
24        args.set(2, "-71.3");
25        args.set(3, "42");
26        args.set(4, "-72");
27        search.action(args);
28        assertEquals(search.getOutput(), "/n/0 -> /n/1 : /w/0\n" +
29          "/n/1 -> /n/2 : /w/1\n" +
30          "/n/2 -> /n/5 : /w/4\n");
31      } catch(Exception e) {
32        System.err.println("ERROR: Error parsing file in testNumLines");
33      }
34    }
35    @Test
36    public void testRouteName() {
37      reader.action(file);
38      try {
39        args.set(1, "\"Sootball Ln\"");
40        args.set(2, "\"Chihiro Ave\"");
41        args.set(3, "\"Sootball Ln\"");
42        args.set(4, "\"Yubaba St\"");
43        search.action(args);
44        assertEquals(search.getOutput(), "/n/1 -> /n/4 : /w/3\n");
45      } catch(Exception e) {
46        System.err.println("ERROR: Error parsing file in testNumLines");
47      }
